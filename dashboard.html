// 1. Satellite Map Base const map = L.map('map').setView([37.5, -79.2], 12); L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}').addTo(map); // 2. High-Performance Red Pen let isDrawing = false; let currentPath = null; map.on('mousedown', (e) => { isDrawing = true; map.dragging.disable(); // Locks map while drawing currentPath = L.polyline([], {color: 'red', weight: 4}).addTo(map); }); map.on('mousemove', (e) => { if (isDrawing) currentPath.addLatLng(e.latlng); }); map.on('mouseup', () => { isDrawing = false; map.dragging.enable(); // Unlocks map }); // 3. Precision Placement System let selectedUnit = null; function setUnit(name, color) { selectedUnit = {name, color}; map.getContainer().style.cursor = 'crosshair'; } map.on('click', (e) => { if (selectedUnit) { const marker = L.circleMarker(e.latlng, { fillColor: selectedUnit.color, fillOpacity: 1, color: '#fff', radius: 8 }).addTo(map); // Right-click to remove marker.on('contextmenu', () => map.removeLayer(marker)); selectedUnit = null; map.getContainer().style.cursor = ''; } });
